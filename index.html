/* 🌌 3D 배경 */ let stars=[]; for(let i=0;i<120;i++){ stars.push({ x:Math.random()*800, y:Math.random()*600, z:Math.random()*3 + 0.5 }); } function draw(){ /* 배경 */ ctx.fillStyle="#05060a"; ctx.fillRect(0,0,800,600); /* 🌌 3D 패럴랙스 */ stars.forEach(s=>{ // 깊이 이동 s.y += s.z * 0.6; if(s.y > 600){ s.y = 0; s.x = Math.random()*800; } // 🔥 핵심: 마우스 패럴랙스 let px = s.x + (mx-400)*0.04*s.z; let py = s.y + (my-300)*0.04*s.z; let size = s.z * 1.5; ctx.fillStyle = "white"; ctx.fillRect(px, py, size, size); }); /* 타이틀 */ ctx.fillStyle="#fff"; ctx.textAlign="center"; ctx.font="bold 70px monospace"; ctx.fillText("CRT 3D BG",400,260); ctx.font="18px monospace"; ctx.fillText("BACKGROUND ONLY",400,320); }
